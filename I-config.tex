%% style
%% title
\definetitlestyle{HWTitleStyle}{
    width=\textwidth, linewidth=15pt, titletotopverticalspace=40pt
}{
    \begin{scope}[line width=\titlelinewidth,]
    \draw[color=titleLine,round cap-round cap]
    (\titleposleft,\titleposbottom)--(\titleposright,\titleposbottom);
    \end{scope}
}

%% theme and style
%% background
\definebackgroundstyle{mystyle}{
    \draw[line width=0pt, bottom color=backgroundcolor, top
     color=backgroundcolor!60!white] (bottomleft) rectangle (topright);
    \draw[draw=none, line width=0pt, bottom color=titlebgcolor, top
     color=framecolor] (bottomleft) rectangle ($(bottomleft)+(\textwidth,3)$);
}


\usebackgroundstyle{mystyle}
\usecolorstyle[colorPalette=HWPalette]{sampleColorStyle}
\usetitlestyle{HWTitleStyle}
